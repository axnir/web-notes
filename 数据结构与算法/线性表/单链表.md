# 单链表


##### 实现
```javascript
/*
    有头结点的版本
    定义带有头结点的单链表的优点：
        使得链表的第一个位置和其他位置的操作统一
        使得空表和非空表的操作统一
*/

//定义node
function Node(value) {
    this.value = value
    this.next = null
}

//定义单链表
function LinkedList() {
    //头节点
    let node = new Node()
    //头指针
    this.head = node
    this.length = 0
}

//从尾部插入节点
LinkedList.prototype.push = function (value) {
    let node = new Node(value)
    //设置p保存头节点
    let p = this.head
    //直到p.next = null,即到最后一个节点
    while (p.next !== null) {
        p = p.next
    }
    p.next = node
    // 长度加1
    this.length++
}

//从头部插入节点
LinkedList.prototype.insert = function (value) {
    let node = new Node(value)
    node.next = this.head.next
    this.head.next = node
    this.length++
}
//按值查找节点
LinkedList.prototype.find = function (value) {
    let p = this.head.next
    while (p !== null && p.value !== value) {
        p = p.next
    }
    return p
}

//按序号查找节点
LinkedList.prototype.getNode = function (i) {
    if (i < 0 || i > this.length - 1 || i === undefined) return
    let p = this.head.next
    for (let j = 0; j < i; j++) {
        p = p.next
    }
    return p
}

// 根据序号删除节点
LinkedList.prototype.delNode = function (i) {
    if (i < 0 || i > this.length - 1 || i === undefined) return
    //temp是p的前驱节点，开始时temp指向p，将temp指向p.next，即temp.next = p.next, p就被从链表中删除了
    let temp = this.head
    for (let j = 0; j < i; j++) {
        temp = p
        p = p.next
    }
    temp.next = p.next
    this.length--
}
```
